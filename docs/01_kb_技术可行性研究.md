**推荐方案**：使用 **OpenXML SDK** (DocumentFormat.OpenXml)

```csharp
// OpenXML SDK 示例：生成带表格的Word文档
using DocumentFormat.OpenXml;
using DocumentFormat.OpenXml.Packaging;
using DocumentFormat.OpenXml.Wordprocessing;

public class WordDocumentGenerator
{
    public void GenerateDocument(string templatePath, Dictionary<string, string> fields, string outputPath)
    {
        // 复制模板
        File.Copy(templatePath, outputPath, true);
        
        using (WordprocessingDocument doc = WordprocessingDocument.Open(outputPath, true))
        {
            // 替换普通字段
            foreach (var field in fields)
            {
                string placeholder = $"{{{field.Key}}}";
                ReplaceTextInDocument(doc, placeholder, field.Value);
            }
            
            doc.Save();
        }
    }
    
    private void ReplaceTextInDocument(WordprocessingDocument doc, string placeholder, string value)
    {
        var body = doc.MainDocumentPart.Document.Body;
        
        foreach (var text in body.Descendants<Text>())
        {
            if (text.Text.Contains(placeholder))
            {
                text.Text = text.Text.Replace(placeholder, value);
            }
        }
    }
}
```

#### 2.2.2 模板解析工具

```csharp
using DocumentFormat.OpenXml.Packaging;
using System.Text.RegularExpressions;

public class TemplateParser
{
    public (List<string> Fields, Dictionary<string, List<string>> Tables) ParseTemplate(string templatePath)
    {
        var fields = new List<string>();
        var tables = new Dictionary<string, List<string>>();
        
        using (WordprocessingDocument doc = WordprocessingDocument.Open(templatePath, false))
        {
            var body = doc.MainDocumentPart.Document.Body;
            
            // 提取普通字段 {字段名}
            foreach (var text in body.Descendants<Text>())
            {
                var matches = Regex.Matches(text.Text, @"\{([^}]+)\}");
                foreach (Match match in matches)
                {
                    string fieldName = match.Groups[1].Value;
                    if (!fieldName.Contains("."))
                    {
                        fields.Add(fieldName);
                    }
                    else
                    {
                        // 表格字段 {表格名.字段名}
                        var parts = fieldName.Split('.');
                        string tableName = parts[0];
                        string columnName = parts[1];
                        
                        if (!tables.ContainsKey(tableName))
                        {
                            tables[tableName] = new List<string>();
                        }
                        if (!tables[tableName].Contains(columnName))
                        {
                            tables[tableName].Add(columnName);
                        }
                    }
                }
            }
        }
        
        return (fields.Distinct().ToList(), tables);
    }
}
```

```csharp
using GitHub.Copilot.SDK;
using GitHub.Copilot.SDK.Models;

public class CopilotAIService
{
    private readonly ICopilotClient _copilotClient;
    private readonly ILogger<CopilotAIService> _logger;
    
    public CopilotAIService(ILogger<CopilotAIService> logger)
    {
        _logger = logger;
        _copilotClient = CopilotClient.Create(new CopilotOptions
        {
            // 使用环境变量或配置
            AccessToken = Environment.GetEnvironmentVariable("GITHUB_COPILOT_TOKEN")
        });
    }
    
    /// <summary>
    /// 从文件中提取结构化数据
    /// </summary>
    public async Task<ExtractionResult> ExtractDataFromFileAsync(
        string filePath, 
        List<FieldDefinition> templateFields)
    {
        // 创建会话
        var session = await _copilotClient.CreateSessionAsync();
        
        // 上传文件
        var fileContent = await File.ReadAllBytesAsync(filePath);
        var uploadedFile = await session.UploadFileAsync(filePath, fileContent);
        
        // 构建提示词
        var prompt = BuildExtractionPrompt(templateFields);
        
        // 发送请求
        var response = await session.SendMessageAsync(new ChatMessage
        {
            Role = MessageRole.User,
            Content = prompt,
            FileIds = new List<string> { uploadedFile.Id }
        });
        
        // 解析结果
        return ParseExtractionResponse(response.Content, templateFields);
    }
    
    /// <summary>
    /// 构建数据提取提示词
    /// </summary>
    private string BuildExtractionPrompt(List<FieldDefinition> fields)
    {
        var sb = new StringBuilder();
        sb.AppendLine("请从上传的文件中提取以下字段的信息，并以JSON格式返回：");
        sb.AppendLine();
        sb.AppendLine("字段列表：");
        foreach (var field in fields)
        {
            sb.AppendLine($"- {field.Name} ({field.Description})");
        }
        sb.AppendLine();
        sb.AppendLine("要求：");
        sb.AppendLine("1. 如果某个字段在文件中找不到对应信息，返回null");
        sb.AppendLine("2. 对于表格数据，返回数组格式");
        sb.AppendLine("3. 返回严格的JSON格式，不要包含其他说明文字");
        sb.AppendLine();
        sb.AppendLine("JSON格式示例：");
        sb.AppendLine("{");
        sb.AppendLine("    \"项目名称\": \"示例项目\",");
        sb.AppendLine("    \"负责人\": \"张三\",");
        sb.AppendLine("    \"成员列表\": [");
        sb.AppendLine("        {\"姓名\": \"李四\", \"职务\": \"工程师\"},");
        sb.AppendLine("        {\"姓名\": \"王五\", \"职务\": \"设计师\"}");
        sb.AppendLine("    ]");
        sb.AppendLine("}");
        
        return sb.ToString();
    }
}
```

```csharp
using Microsoft.AgentFramework;

public class CopilotDataExtractionTool : ITool

                Message = $"成功提取 {result.Fields.Count} 个字段"
            };
        }
        catch (Exception ex)
        {
            return new ToolResult
            {
                Success = false,
                Error = ex.Message
            };
        }
    }
}
```

## 四、Word 复杂元素处理技术可行性研究

### 4.1 概述

Word 文档中除了普通文本和表格外，还可能包含以下复杂元素：
- **内容控件**（Content Controls）：纯文本、富文本、下拉框、复选框、日期选择器等
- **图片**：Logo、签名、照片等
- **表单域**（Legacy Form Fields）：旧版表单字段
- **书签**（Bookmarks）：用于定位的标记

### 4.2 内容控件（Content Controls）处理

内容控件是 Word 2007+ 推荐的结构化文档元素，适合作为模板占位符。

#### 4.2.1 内容控件类型

| 控件类型 | 用途 | OpenXML SDK 类 |
|----------|------|----------------|
| 纯文本 | 无格式文本 | SdtContentText |
| 富文本 | 带格式文本 | SdtContentText |
| 下拉列表 | 选择预定义值 | SdtContentDropDownList |
| 组合框 | 选择或输入 | SdtContentComboBox |
| 复选框 | 是/否选择 | SdtContentCheckBox |
| 日期选择器 | 日期输入 | SdtContentDate |
| 图片 | 图像占位符 | SdtContentPicture |

#### 4.2.2 内容控件填充实现

```csharp
using DocumentFormat.OpenXml.Packaging;
using DocumentFormat.OpenXml.Wordprocessing;

public class ContentControlFiller
{
    /// <summary>
    /// 填充纯文本内容控件
    /// </summary>
    public void FillTextControl(WordprocessingDocument doc, string tag, string value)
    {
        var sdtElements = doc.MainDocumentPart.Document.Body
            .Descendants<SdtRun>()
            .Where(sdt => {
                var tagElement = sdt.SdtProperties?.GetFirstChild<Tag>();
                return tagElement?.Val == tag;
            });
        
        foreach (var sdt in sdtElements)
        {
            var content = sdt.GetFirstChild<SdtContentRun>();
            if (content != null)
            {
                var text = content.GetFirstChild<Run>()?.GetFirstChild<Text>();
                if (text != null)
                {
                    text.Text = value;
                }
            }
        }
    }
    
    /// <summary>
    /// 填充下拉列表内容控件
    /// </summary>
    public void FillDropDownControl(WordprocessingDocument doc, string tag, string selectedValue)
    {
        var sdtElements = doc.MainDocumentPart.Document.Body
            .Descendants<SdtRun>()
            .Where(sdt => {
                var tagElement = sdt.SdtProperties?.GetFirstChild<Tag>();
                return tagElement?.Val == tag;
            });
        
        foreach (var sdt in sdtElements)
        {
            var dropDown = sdt.SdtProperties?.GetFirstChild<SdtContentDropDownList>();
            if (dropDown != null)
            {
                // 检查值是否在列表中
                var exists = dropDown.Elements<ListItem>()
                    .Any(item => item.DisplayText == selectedValue);
                
                if (exists)
                {
                    var content = sdt.GetFirstChild<SdtContentRun>();
                    var text = content?.GetFirstChild<Run>()?.GetFirstChild<Text>();
                    if (text != null)
                    {
                        text.Text = selectedValue;
                    }
                }
            }
        }
    }
    
    /// <summary>
    /// 填充复选框内容控件
    /// </summary>
    public void FillCheckBoxControl(WordprocessingDocument doc, string tag, bool isChecked)
    {
        var sdtElements = doc.MainDocumentPart.Document.Body
            .Descendants<SdtRun>()
            .Where(sdt => {
                var tagElement = sdt.SdtProperties?.GetFirstChild<Tag>();
                return tagElement?.Val == tag;
            });
        
        foreach (var sdt in sdtElements)
        {
            var checkBox = sdt.SdtProperties?.GetFirstChild<SdtContentCheckBox>();
            if (checkBox != null)
            {
                // 设置复选框状态
                var checkedElement = checkBox.GetFirstChild<Checked>();
                if (checkedElement != null)
                {
                    checkedElement.Val = isChecked ? OnOffValue.FromBoolean(true) : OnOffValue.FromBoolean(false);
                }
            }
        }
    }
}
```

### 4.3 图片替换

#### 4.3.1 图片内容控件替换

```csharp
using DocumentFormat.OpenXml.Packaging;
using DocumentFormat.OpenXml.Wordprocessing;
using DocumentFormat.OpenXml.Drawing;
using A = DocumentFormat.OpenXml.Drawing;
using DW = DocumentFormat.OpenXml.Drawing.Wordprocessing;
using PIC = DocumentFormat.OpenXml.Drawing.Pictures;

public class PictureReplacer
{
    /// <summary>
    /// 替换图片内容控件中的图片
    /// </summary>
    public void ReplacePictureInControl(
        WordprocessingDocument doc, 
        string tag, 
        string newImagePath)
    {
        // 查找图片内容控件
        var sdtBlocks = doc.MainDocumentPart.Document.Body
            .Descendants<SdtBlock>()
            .Where(sdt => {
                var tagElement = sdt.SdtProperties?.GetFirstChild<Tag>();
                var pictureContent = sdt.SdtProperties?.GetFirstChild<SdtContentPicture>();
                return tagElement?.Val == tag && pictureContent != null;
            });
        
        foreach (var sdtBlock in sdtBlocks)
        {
            // 获取Drawing元素
            var drawing = sdtBlock.Descendants<Drawing>().FirstOrDefault();
            if (drawing != null)
            {
                // 获取图片引用ID
                var blip = drawing.Descendants<A.Blip>().FirstOrDefault();
                if (blip != null)
                {
                    string embedId = blip.Embed;
                    
                    // 获取ImagePart
                    var imagePart = doc.MainDocumentPart.GetPartById(embedId) as ImagePart;
                    if (imagePart != null)
                    {
                        // 替换图片内容
                        using (FileStream stream = new FileStream(newImagePath, FileMode.Open))
                        {
                            imagePart.FeedData(stream);
                        }
                    }
                }
            }
        }
    }
    
    /// <summary>
    /// 在指定位置插入新图片
    /// </summary>
    public void InsertPicture(WordprocessingDocument doc, string bookmarkName, string imagePath)
    {
        var mainPart = doc.MainDocumentPart;
        ImagePart imagePart = mainPart.AddImagePart(ImagePartType.Jpeg);
        
        using (FileStream stream = new FileStream(imagePath, FileMode.Open))
        {
            imagePart.FeedData(stream);
        }
        
        // 创建Drawing元素
        Drawing drawing = CreateDrawingElement(mainPart.GetIdOfPart(imagePart), 4000000, 3000000);
        
        // 在书签位置插入
        var bookmark = mainPart.Document.Body
            .Descendants<BookmarkStart>()
            .FirstOrDefault(b => b.Name == bookmarkName);
        
        if (bookmark != null)
        {
            var run = new Run(drawing);
            bookmark.Parent.InsertAfterSelf(run);
        }
    }
    
    private Drawing CreateDrawingElement(string relationshipId, long width, long height)
    {
        return new Drawing(
            new DW.Inline(
                new DW.Extent { Cx = width, Cy = height },
                new DW.EffectExtent { LeftEdge = 0, TopEdge = 0, RightEdge = 0, BottomEdge = 0 },
                new DW.DocProperties { Id = 1, Name = "Picture" },
                new DW.NonVisualGraphicFrameDrawingProperties(
                    new A.GraphicFrameLocks { NoChangeAspect = true }),
                new A.Graphic(
                    new A.GraphicData(
                        new PIC.Picture(
                            new PIC.NonVisualPictureProperties(
                                new A.ShapeLocks { NoChangeAspect = true }),
                            new PIC.BlipFill(new A.Blip { Embed = relationshipId }),
                            new PIC.ShapeProperties(
                                new A.Transform2D(
                                    new A.Offset { X = 0, Y = 0 },
                                    new A.Extents { Cx = width, Cy = height })))
                    { Uri = "http://schemas.openxmlformats.org/drawingml/2006/picture" }))
            { DistanceFromTop = 0, DistanceFromBottom = 0, DistanceFromLeft = 0, DistanceFromRight = 0 });
    }
}
```

```csharp
            .Descendants<CheckBox>()
            .Where(c => {

                return name?.Val == fieldName;
            });
```

### 4.5 综合文档生成器

```csharp
public class AdvancedDocumentGenerator
{
    private readonly ContentControlFiller _contentControlFiller;
    private readonly PictureReplacer _pictureReplacer;
    private readonly LegacyFormFieldFiller _legacyFormFieldFiller;
    
    /// <summary>
    /// 生成文档（支持复杂元素）
    /// </summary>
    public void GenerateDocument(
        string templatePath, 
        DocumentData data, 
        string outputPath)
    {
        // 复制模板
        File.Copy(templatePath, outputPath, true);
        
        using (WordprocessingDocument doc = WordprocessingDocument.Open(outputPath, true))
        {
            // 1. 填充普通文本内容控件
            foreach (var textField in data.TextFields)
            {
                _contentControlFiller.FillTextControl(doc, textField.Key, textField.Value);
            }
            
            // 2. 填充下拉列表
            foreach (var dropDown in data.DropDownFields)
            {
                _contentControlFiller.FillDropDownControl(doc, dropDown.Key, dropDown.Value);
            }
            
            // 3. 填充复选框
            foreach (var checkBox in data.CheckBoxFields)
            {
                _contentControlFiller.FillCheckBoxControl(doc, checkBox.Key, checkBox.Value);
            }
            
            // 4. 替换图片
            foreach (var picture in data.Pictures)
            {
                _pictureReplacer.ReplacePictureInControl(doc, picture.Key, picture.Value);
            }
            
            // 5. 填充旧版表单域
            foreach (var legacyField in data.LegacyTextFields)
            {
                _legacyFormFieldFiller.FillTextFormField(doc, legacyField.Key, legacyField.Value);
            }
            
            // 6. 填充表格
            FillTables(doc, data.Tables);
            
            doc.Save();
        }
    }
}
```

```csharp
public class FieldMatcher
{
    public async Task<Dictionary<string, object>> MatchFieldsAsync(
        Dictionary<string, object> extractedData,
        List<FieldDefinition> templateFields)
    {
        var matchedData = new Dictionary<string, object>();
        
        foreach (var templateField in templateFields)
        {
            // 1. 精确匹配
            if (extractedData.ContainsKey(templateField.Name))
            {
                matchedData[templateField.Name] = extractedData[templateField.Name];
                continue;
            }
            
            // 2. 同义词匹配
            var synonymMatch = await MatchBySynonymsAsync(
                templateField, 
                extractedData);
            if (synonymMatch != null)
            {
                matchedData[templateField.Name] = synonymMatch;
                continue;
            }
            
            // 3. LLM语义匹配（对于复杂情况）
            var semanticMatch = await MatchByLLMAsync(
                templateField, 
                extractedData);
            if (semanticMatch != null)
            {
                matchedData[templateField.Name] = semanticMatch;
            }
        }
        
        return matchedData;
    }
}
```
